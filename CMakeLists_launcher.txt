cmake_minimum_required(VERSION 3.16)

project(melonDS-launcher)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Cross-compilation settings (same as main build)
set(CROSS_PATH "/home/corey/x-tools/aarch64-buildroot-linux-gnu")
set(CROSS_PREFIX "aarch64-buildroot-linux-gnu")

set(CMAKE_SYSTEM_NAME Linux)
set(CMAKE_SYSTEM_PROCESSOR aarch64)
set(CMAKE_C_COMPILER "${CROSS_PATH}/bin/${CROSS_PREFIX}-gcc")
set(CMAKE_CXX_COMPILER "${CROSS_PATH}/bin/${CROSS_PREFIX}-g++")

# Optimization flags for handheld
set(CMAKE_C_FLAGS "-march=armv8-a -mtune=cortex-a53 -O3 -flto -ffast-math -pipe -fomit-frame-pointer -DNDEBUG")
set(CMAKE_CXX_FLAGS "-march=armv8-a -mtune=cortex-a53 -O3 -flto -ffast-math -pipe -fomit-frame-pointer -DNDEBUG")
set(CMAKE_EXE_LINKER_FLAGS "-flto -Wl,--gc-sections -Wl,--strip-all")

# Find SDL2
find_package(PkgConfig REQUIRED)
pkg_check_modules(SDL2 REQUIRED IMPORTED_TARGET sdl2)

# Find required packages
find_package(Threads REQUIRED)

# Include directories - only what we need for the launcher
include_directories(
    ${CMAKE_CURRENT_SOURCE_DIR}/../src
    ${CMAKE_CURRENT_SOURCE_DIR}/..
)

# Platform-specific settings
add_definitions(-DLOW_MEMORY_BUILD)

# Add required definitions
add_definitions(-DMELONDS_VERSION_BASE="1.0")
add_definitions(-DJIT_ENABLED)
add_definitions(-DGDBSTUB_ENABLED)

# Create the launcher executable
add_executable(melonDS-launcher 
    ${CMAKE_CURRENT_SOURCE_DIR}/../melonds_launcher.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/../platform_sdl.cpp
)

# Link with the core library and dependencies
target_link_libraries(melonDS-launcher
    ${CMAKE_CURRENT_SOURCE_DIR}/../build-handheld/src/libcore.a
    ${CMAKE_CURRENT_SOURCE_DIR}/../build-handheld/src/teakra/src/libteakra.a
    PkgConfig::SDL2
    Threads::Threads
    rt
    dl
    m
)

# Set output name
set_target_properties(melonDS-launcher PROPERTIES OUTPUT_NAME "melonDS")
